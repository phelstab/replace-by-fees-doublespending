== Installation

[source,bash]
----
python -m venv venv
pip install -r requirements.txt
----

== Usage

[source,python]
----
# Dryrun / Testrun
python main.py "address" 0.001 -t -n -d 30 --fee1 0.000014 --fee2 0.0014 --optinrbf 

# Real run (remove use -n)
python main.py "address" 0.001 -t -d 30 --fee1 0.000014 --fee2 0.0014 --optinrbf 
----
